`; }).join(''); els.readerContent.innerHTML = html; els.readerContent.style.fontSize = state.fontSize + 'px'; recalculateRenderedLineWeights(); const newIndex = findBestRestoreIndex(restoreMarker || {}); stopPlayback(); setActiveLine(newIndex, true, { force: true, durationMs: 0 }); updateProgress(); }
